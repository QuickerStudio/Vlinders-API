## 🎯 核心洞察

基于对 Vlinder-chat 客户端代码的深度分析，我们发现 Copilot API 不仅仅是简单的聊天 API，而是一个**复杂的 Multi-Agent 编排系统**。

### 关键发现

1. **Agent 不是单一的 AI**，而是一个分层的协作系统
2. **工具调用循环**是核心机制，支持多轮迭代
3. **子 Agent** 可以并行执行，提高效率
4. **上下文管理**非常复杂，包括压缩、缓存、摘要
5. **Hooks 系统**允许在 Agent 生命周期中插入自定义逻辑

#### 1. Agent 编排引擎

```python
# vlinders_server/agent/orchestrator.py

class AgentOrchestrator:
    """Agent 编排器 - 管理 Agent 生命周期和协作"""

    async def invoke_agent(
        self,
        agent_type: AgentType,
        instruction: str,
        context: AgentContext,
        config: AgentConfig
    ) -> AgentResult:
        """调用主 Agent"""

        # 1. 创建 Agent 实例
        agent = self.agent_factory.create(agent_type, config)

        # 2. 初始化上下文
        agent_context = await self.context_manager.create_context(
            instruction=instruction,
            workspace=context.workspace,
            history=context.history
        )

        # 3. 执行工具调用循环
        result = await self.tool_calling_loop.run(
            agent=agent,
            context=agent_context,
            max_iterations=config.max_iterations
        )

        # 4. 处理子 Agent 调用
        if result.subagent_calls:
            subagent_results = await self.execute_subagents_parallel(
                result.subagent_calls
            )
            result.merge_subagent_results(subagent_results)

        # 5. 压缩上下文（如果需要）
        if agent_context.should_compress():
            await self.context_manager.compress_background(agent_context)

        return result

    async def execute_subagents_parallel(
        self,
        subagent_calls: List[SubagentCall]
    ) -> List[SubagentResult]:
        """并行执行多个子 Agent"""

        tasks = [
            self.invoke_subagent(call)
            for call in subagent_calls
        ]

        return await asyncio.gather(*tasks)
```

#### 2. 工具调用循环

```python
# vlinders_server/agent/tool_calling_loop.py

class ToolCallingLoop:
    """工具调用循环 - 核心执行引擎"""

    async def run(
        self,
        agent: Agent,
        context: AgentContext,
        max_iterations: int = 10
    ) -> LoopResult:
        """执行工具调用循环"""

        iteration = 0
        tool_call_rounds = []

        while iteration < max_iterations:
            # 1. 构建 Prompt
            prompt = await self.prompt_builder.build(
                agent=agent,
                context=context,
                available_tools=agent.tools,
                history=tool_call_rounds
            )

            # 2. 调用 LLM
            response = await self.model_service.generate(
                prompt=prompt,
                model=agent.model,
                stream=True
            )

            # 3. 解析工具调用
            tool_calls = self.parse_tool_calls(response)

            if not tool_calls:
                # 没有工具调用，结束循环
                break

            # 4. 执行工具（并行）
            tool_results = await self.execute_tools_parallel(tool_calls)

            # 5. 记录本轮
            tool_call_rounds.append({
                'iteration': iteration,
                'tool_calls': tool_calls,
                'tool_results': tool_results,
                'response': response
            })

            # 6. 更新上下文
            context.add_tool_round(tool_call_rounds[-1])

            iteration += 1

        return LoopResult(
            final_response=response,
            tool_call_rounds=tool_call_rounds,
            iterations=iteration
        )
```

#### 3. 上下文管理引擎

```python
# vlinders_server/context/manager.py

class ContextManager:
    """上下文管理器 - 处理对话历史和压缩"""

    async def compress_background(
        self,
        context: AgentContext
    ) -> CompressionResult:
        """后台压缩对话历史"""

        # 1. 识别可压缩的轮次
        compressible_rounds = self.identify_compressible_rounds(
            context.history
        )

        # 2. 使用小模型生成摘要
        summaries = await self.summarizer.summarize_parallel(
            rounds=compressible_rounds,
            model='claude-haiku-4.5'  # 快速小模型
        )

        # 3. 替换原始内容
        for round_id, summary in summaries.items():
            context.replace_round_with_summary(round_id, summary)

        # 4. 更新 Token 计数
        context.recalculate_tokens()

        return CompressionResult(
            original_tokens=context.original_tokens,
            compressed_tokens=context.current_tokens,
            compression_ratio=context.compression_ratio
        )

    def should_compress(self, context: AgentContext) -> bool:
        """判断是否需要压缩"""

        token_usage_ratio = context.current_tokens / context.max_tokens

        # 使用率 >= 75% 时触发后台压缩
        return token_usage_ratio >= 0.75
```

#### 4. 工具执行引擎

```python
# vlinders_server/tools/executor.py

class ToolExecutor:
    """工具执行器 - 在沙箱中执行工具"""

    async def execute(
        self,
        tool_name: str,
        arguments: Dict[str, Any],
        context: ExecutionContext
    ) -> ToolResult:
        """执行单个工具"""

        # 1. 获取工具实现
        tool = self.tool_registry.get(tool_name)

        # 2. 验证参数
        validated_args = tool.validate_arguments(arguments)

        # 3. 检查权限
        if not self.permission_checker.check(tool, context.user):
            raise PermissionError(f"User not allowed to use {tool_name}")

        # 4. 在沙箱中执行
        result = await self.sandbox.execute(
            tool=tool,
            arguments=validated_args,
            timeout=tool.timeout,
            resource_limits=tool.resource_limits
        )

        # 5. 记录执行日志
        await self.logger.log_tool_execution(
            tool_name=tool_name,
            arguments=arguments,
            result=result,
            duration=result.duration
        )

        return result

    async def execute_parallel(
        self,
        tool_calls: List[ToolCall]
    ) -> List[ToolResult]:
        """并行执行多个工具"""

        tasks = [
            self.execute(call.name, call.arguments, call.context)
            for call in tool_calls
        ]

        return await asyncio.gather(*tasks, return_exceptions=True)
```

#### 5. 代码分析引擎

```python
# vlinders_server/code_analysis/engine.py

class CodeAnalysisEngine:
    """代码分析引擎 - 使用 Tree-sitter 解析代码"""

    async def analyze_file(
        self,
        file_path: str,
        language: str
    ) -> FileAnalysis:
        """分析单个文件"""

        # 1. 读取文件内容
        content = await self.file_reader.read(file_path)

        # 2. 使用 Tree-sitter 解析
        tree = self.parser.parse(content, language)

        # 3. 提取符号
        symbols = self.symbol_extractor.extract(tree)

        # 4. 分析依赖
        dependencies = self.dependency_analyzer.analyze(tree)

        # 5. 生成嵌入
        embeddings = await self.embedding_service.generate(
            code=content,
            symbols=symbols
        )

        return FileAnalysis(
            file_path=file_path,
            language=language,
            symbols=symbols,
            dependencies=dependencies,
            embeddings=embeddings
        )

    async def search_code(
        self,
        query: str,
        workspace: str,
        file_pattern: str = "*"
    ) -> List[CodeMatch]:
        """语义搜索代码"""

        # 1. 生成查询嵌入
        query_embedding = await self.embedding_service.generate(query)

        # 2. 在向量数据库中搜索
        matches = await self.vector_db.search(
            embedding=query_embedding,
            workspace=workspace,
            file_pattern=file_pattern,
            top_k=10
        )

        # 3. 重新排序（考虑更多因素）
        ranked_matches = self.ranker.rank(
            matches=matches,
            query=query,
            factors=['semantic_similarity', 'recency', 'popularity']
        )

        return ranked_matches
```

## 💡 关键设计决策

### 1. 为什么需要三层？

**Vlinder-chat (客户端)**:
- 必须在本地运行（VSCode 扩展）
- 需要访问本地文件系统
- 负责 UI 和用户交互

**Vlinders-API (协议层)**:
- 提供标准化的 API 接口
- 兼容多种客户端（VSCode、Web、CLI）
- 边缘计算，低延迟
- 独立于服务实现，易于版本管理

**Vlinders-Server (服务层)**:
- 复杂的 Agent 编排逻辑
- 需要强大的计算资源
- 需要访问向量数据库、代码索引
- 可以独立扩展和优化

### 2. Vlinders-API 的定位

**不是**:
- ❌ 简单的反向代理
- ❌ 完整的业务逻辑实现

**而是**:
- ✅ 协议定义和转换层
- ✅ 认证和授权网关
- ✅ 请求路由和负载均衡
- ✅ API 文档和 SDK 生成器

### 3. 通信协议选择

**客户端 ↔ API**:
- HTTPS + JSON (标准 REST API)
- Server-Sent Events (流式响应)
- 兼容 OpenAI/Anthropic 格式
