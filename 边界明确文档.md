#### 1. 模型推理服务（基于 vLLM）

```python
# vlinders_server/inference/vllm_service.py

from vllm import LLM, SamplingParams
from vllm.engine.arg_utils import AsyncEngineArgs
from vllm.engine.async_llm_engine import AsyncLLMEngine

class VLLMInferenceService:
    """基于 vLLM 的模型推理服务"""

    def __init__(self):
        self.engines = {}  # 多模型支持

    async def load_model(self, model_name: str, model_path: str):
        """加载模型到 vLLM"""
        engine_args = AsyncEngineArgs(
            model=model_path,
            tensor_parallel_size=4,  # 多 GPU 并行
            dtype="float16",
            max_model_len=32768,
            gpu_memory_utilization=0.9
        )

        engine = AsyncLLMEngine.from_engine_args(engine_args)
        self.engines[model_name] = engine

    async def generate(
        self,
        model: str,
        prompt: str,
        max_tokens: int = 2048,
        temperature: float = 0.7,
        stream: bool = False
    ):
        """生成文本"""
        engine = self.engines.get(model)
        if not engine:
            raise ValueError(f"Model {model} not loaded")

        sampling_params = SamplingParams(
            temperature=temperature,
            max_tokens=max_tokens,
            top_p=0.95
        )

        if stream:
            # 流式生成
            async for output in engine.generate(prompt, sampling_params):
                yield output.outputs[0].text
        else:
            # 非流式生成
            results = await engine.generate(prompt, sampling_params)
            return results[0].outputs[0].text
```

#### 2. Agent 编排引擎

```python
# vlinders_server/agent/orchestrator.py

class AgentOrchestrator:
    """Agent 编排引擎 - 管理 Agent 生命周期"""

    def __init__(
        self,
        inference_service: VLLMInferenceService,
        tool_executor: ToolExecutor,
        context_manager: ContextManager
    ):
        self.inference = inference_service
        self.tools = tool_executor
        self.context = context_manager

    async def run_agent(
        self,
        agent_type: str,
        instruction: str,
        context: Dict,
        max_iterations: int = 10
    ):
        """运行 Agent"""

        # 1. 初始化 Agent 上下文
        agent_context = await self.context.create(
            instruction=instruction,
            workspace=context.get('workspace'),
            history=context.get('history', [])
        )

        # 2. 工具调用循环
        iteration = 0
        while iteration < max_iterations:
            # 构建 Prompt
            prompt = self.build_prompt(agent_context)

            # 调用模型
            response = await self.inference.generate(
                model=agent_context.model,
                prompt=prompt,
                stream=False
            )

            # 解析工具调用
            tool_calls = self.parse_tool_calls(response)

            if not tool_calls:
                break  # 没有工具调用，结束

            # 执行工具
            tool_results = await self.tools.execute_batch(tool_calls)

            # 更新上下文
            agent_context.add_round(tool_calls, tool_results)

            iteration += 1

        return {
            'response': response,
            'iterations': iteration,
            'tool_calls': agent_context.tool_calls
        }
```

#### 3. 工具执行引擎

```python
# vlinders_server/tools/executor.py

class ToolExecutor:
    """工具执行引擎"""

    def __init__(self):
        self.tools = {}
        self.register_builtin_tools()

    def register_builtin_tools(self):
        """注册内置工具"""
        self.tools['search_code'] = CodeSearchTool()
        self.tools['read_file'] = ReadFileTool()
        self.tools['analyze_code'] = CodeAnalysisTool()
        self.tools['semantic_search'] = SemanticSearchTool()

    async def execute(self, tool_name: str, arguments: Dict):
        """执行单个工具"""
        tool = self.tools.get(tool_name)
        if not tool:
            raise ValueError(f"Tool {tool_name} not found")

        return await tool.execute(arguments)

    async def execute_batch(self, tool_calls: List[ToolCall]):
        """批量执行工具（并行）"""
        tasks = [
            self.execute(call.name, call.arguments)
            for call in tool_calls
        ]
        return await asyncio.gather(*tasks)
```

#### 4. 代码分析引擎

```python
# vlinders_server/code_analysis/engine.py

import tree_sitter
from tree_sitter_languages import get_language, get_parser

class CodeAnalysisEngine:
    """代码分析引擎 - 基于 Tree-sitter"""

    def __init__(self):
        self.parsers = {}
        self.init_parsers()

    def init_parsers(self):
        """初始化各语言解析器"""
        for lang in ['python', 'javascript', 'typescript', 'go', 'rust']:
            self.parsers[lang] = get_parser(lang)

    def parse_file(self, code: str, language: str):
        """解析代码文件"""
        parser = self.parsers.get(language)
        if not parser:
            raise ValueError(f"Language {language} not supported")

        tree = parser.parse(bytes(code, 'utf8'))
        return tree

    def extract_symbols(self, tree, language: str):
        """提取符号（函数、类、变量等）"""
        symbols = []

        # 使用 Tree-sitter 查询
        query = self.get_symbol_query(language)
        captures = query.captures(tree.root_node)

        for node, name in captures:
            symbols.append({
                'name': node.text.decode('utf8'),
                'type': name,
                'start_line': node.start_point[0],
                'end_line': node.end_point[0]
            })

        return symbols
```

```python
# vlinders_server/api/internal.py

from fastapi import FastAPI, Header, HTTPException

app = FastAPI()

INTERNAL_SECRET = os.getenv('INTERNAL_SECRET')

def verify_internal_auth(x_internal_auth: str = Header(...)):
    """验证内部请求"""
    if x_internal_auth != INTERNAL_SECRET:
        raise HTTPException(status_code=403, detail="Forbidden")

@app.post("/internal/chat")
async def internal_chat(
    request: InternalChatRequest,
    auth: str = Depends(verify_internal_auth)
):
    """内部聊天接口 - 只接受来自 Vlinders-API 的请求"""

    # 1. 运行 Agent
    result = await agent_orchestrator.run_agent(
        agent_type='chat',
        instruction=request.messages[-1]['content'],
        context={
            'messages': request.messages,
            'tools': request.tools
        }
    )

    # 2. 返回结果
    return {
        'id': generate_id(),
        'model': request.model,
        'choices': [{
            'message': {
                'role': 'assistant',
                'content': result['response']
            },
            'finish_reason': 'stop'
        }],
        'usage': {
            'prompt_tokens': result['prompt_tokens'],
            'completion_tokens': result['completion_tokens'],
            'total_tokens': result['total_tokens']
        }
    }

@app.post("/internal/agent")
async def internal_agent(
    request: InternalAgentRequest,
    auth: str = Depends(verify_internal_auth)
):
    """内部 Agent 接口"""

    result = await agent_orchestrator.run_agent(
        agent_type=request.agent_type,
        instruction=request.instruction,
        context=request.context,
        max_iterations=request.max_iterations
    )

    return result
```
